import os
from qwen_agent.agents import Assistant

# 1. Configure the Local Model
# Replace with the actual path to your downloaded Qwen3-1.7B directory
local_model_path = "./models/Qwen3-1.7B-Instruct"

llm_cfg = {
    'model': local_model_path,
    'model_type': 'transformers',  # Specifies using the local transformers backend
    'generate_cfg': {
        'thought_in_content': True, # Enables Qwen3's native thinking mode
        'max_new_tokens': 1024,
        'temperature': 0.6          # Recommended for Qwen3 thinking mode
    }
}

# 2. Define MCP Tools
# This connects the agent to a standard MCP server (e.g., local filesystem)
tools = [
    {
        'mcpServers': {
            'filesystem': {
                'command': 'uvx',
                'args': ['mcp-server-filesystem', '/path/to/allowed/directory']
            }
        }
    },
    'code_interpreter', # Built-in Qwen-Agent tool
]

# 3. Initialize the Agent
bot = Assistant(llm=llm_cfg, function_list=tools)

# 4. Run a Task
# The agent will automatically decide whether to "think" and which tool to use
messages = [{'role': 'user', 'content': 'Read the file "data.txt" in my allowed directory and summarize it.'}]

print("--- Agent is processing ---")
for responses in bot.run(messages=messages):
    # This handles streaming output, including <think> blocks if generated
    pass

print(responses[-1]['content'])
