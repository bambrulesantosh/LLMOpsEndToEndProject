import asyncio
import torch
from transformers import AutoModelForCausalLM, AutoTokenizer
from mcp import ClientSession, StdioServerParameters
from mcp.client.stdio import stdio_client

# 1. Load Local Qwen3-1.7B
model_path = "./path/to/your/Qwen3-1.7B-Instruct"
tokenizer = AutoTokenizer.from_pretrained(model_path)
model = AutoModelForCausalLM.from_pretrained(
    model_path,
    torch_dtype=torch.bfloat16,
    device_map="auto"
).eval()

async def run_mcp_agent():
    # 2. Configure the MCP Server (Example: a simple local math server)
    server_params = StdioServerParameters(
        command="python",
        args=["math_server.py"] # Replace with your actual server script or command
    )

    async with stdio_client(server_params) as (read, write):
        async with ClientSession(read, write) as session:
            # Initialize connection and list tools
            await session.initialize()
            available_tools = await session.list_tools()
            
            # 3. Create a System Prompt including the tool schemas
            tool_schemas = [f"Name: {t.name}, Description: {t.description}" for t in available_tools.tools]
            system_prompt = f"You have access to these tools:\n{tool_schemas}\nRespond in JSON format to call a tool: {{\"tool\": \"name\", \"args\": {{}}}}"

            # 4. Model Inference
            user_query = "What is 15.5 plus 24.5?"
            messages = [
                {"role": "system", "content": system_prompt},
                {"role": "user", "content": user_query}
            ]

            # Apply Qwen3 chat template
            input_text = tokenizer.apply_chat_template(messages, tokenize=False, add_generation_prompt=True)
            inputs = tokenizer(input_text, return_tensors="pt").to(model.device)

            with torch.no_grad():
                output_ids = model.generate(**inputs, max_new_tokens=256)
                response = tokenizer.decode(output_ids[0][inputs.input_ids.shape[1]:], skip_special_tokens=True)

            print(f"Model Output: {response}")

            # 5. Execute Tool if requested (Basic Parsing)
            # In a real app, you would parse the JSON and call:
            # result = await session.call_tool("tool_name", {"arg": "val"})
            # print(f"Tool Result: {result}")

if __name__ == "__main__":
    asyncio.run(run_mcp_agent())

import asyncio
from transformers import AutoTokenizer, AutoModelForCausalLM, pipeline
from langchain_huggingface import ChatHuggingFace, HuggingFacePipeline
from langchain_mcp_adapters.tools import load_mcp_tools
from langgraph.prebuilt import create_react_agent
from mcp import ClientSession, StdioServerParameters
from mcp.client.stdio import stdio_client

# 1. CPU Model Configuration
model_path = "./path/to/gpt-oss-20b"

print("--- Loading 20B Model to CPU (This will take time and ~42GB RAM) ---")
tokenizer = AutoTokenizer.from_pretrained(model_path)
model = AutoModelForCausalLM.from_pretrained(
    model_path,
    device_map="cpu",               # Forces everything to System RAM
    low_cpu_mem_usage=True,         # Reduces RAM spikes during loading
    torch_dtype="auto",             # Uses the model's native precision
    trust_remote_code=True
)

# 2. Build the CPU Pipeline
# Note: Inference on CPU is slow; max_new_tokens is kept low for responsiveness
hf_pipe = HuggingFacePipeline(
    pipeline=pipeline(
        "text-generation",
        model=model,
        tokenizer=tokenizer,
        max_new_tokens=256,
        temperature=0.1
    )
)

llm = ChatHuggingFace(llm=hf_pipe)

# 3. Setup MCP Server (Using Fetch server to read web content)
server_params = StdioServerParameters(
    command="uvx",
    args=["mcp-server-fetch"]
)

async def run_cpu_agent():
    async with stdio_client(server_params) as (read, write):
        async with ClientSession(read, write) as session:
            await session.initialize()
            
            # 4. Fetch tools from MCP and convert for LangChain
            tools = await load_mcp_tools(session)
            
            # 5. Create Agent
            agent = create_react_agent(llm, tools)
            
            # 6. Execute Task
            print("--- Agent Ready ---")
            query = {"messages": [("user", "Fetch the content of https://example.com and summarize it.")]}
            
            async for chunk in agent.astream(query):
                for node, values in chunk.items():
                    print(f"[{node}]: {values['messages'][-1].content}")

if __name__ == "__main__":
    asyncio.run(run_cpu_agent())
